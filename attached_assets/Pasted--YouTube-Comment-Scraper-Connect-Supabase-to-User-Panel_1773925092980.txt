### Supabase project

- Function name: fetch-comments (NOT scrape-comments — use this exact name everywhere)

### Edge function behavior (CRITICAL — read carefully)
The fetch-comments function works differently from a simple one-shot scrape.
It uses a CHUNKED invocation model:

1. First call: pass { videoUrl, maxComments, filters }
   → Returns { done, nextPageToken, job_id, comment_count, credits_used, balance_after }

2. If done=false: call AGAIN with { videoUrl, maxComments, filters, jobId, pageToken: nextPageToken }
   → Repeats until done=true

The frontend MUST loop these calls until done=true.
Each call fetches up to 2,000 comments (PER_INVOCATION_LIMIT).
So a 10,000 comment job requires ~5 chained calls from the frontend.

### Edge function request body shape
```typescript
{
  videoUrl:    string,          // full YouTube URL
  maxComments: number | null,   // null = fetch ALL comments
  filters: {
    sortBy?:         "relevance" | "newest",
    minLikes?:       number,
    dateFrom?:       string,    // ISO date string
    keyword?:        string,    // comma-separated keywords
    includeReplies?: boolean,
    maxComments?:    number,
  },
  jobId?:     string,           // only on 2nd+ calls (resume)
  pageToken?: string,           // only on 2nd+ calls (from previous response)
}
```

### Edge function response shape
```typescript
{
  success:       boolean,
  done:          boolean,         // true = all pages fetched
  nextPageToken: string | null,   // null when done=true
  job_id:        string,
  video_title:   string,
  channel_name:  string,
  thumbnail:     string,
  comment_count: number,          // ACCUMULATED total across all invocations
  credits_used:  number,
  balance_after: number,
  cancelled:     boolean,
}
```

### Error response for insufficient credits
```typescript
{
  error:          "insufficient_credits",
  message:        "Not enough credits. Please top up.",
  balance:        number,
  credits_needed: number | string,
}
```

### Credit system
- Credits are deducted PER COMMENT via atomic_credit_deduct() PostgreSQL function
- 1 credit = 1 comment (NOT 1 credit = 1000 comments)
- Balance = SUM of all rows in credit_ledger for user
- source_type values allowed: "purchase" | "admin_grant" | "refund" | "job_reserve"
- The edge function handles ALL credit deduction — frontend does NOT insert credit_ledger rows

### Database tables (confirmed exist)
- profiles, jobs, comments, credit_ledger, credit_packages
- platform_settings (has key: free_plan_credits)
- notifications, youtube_api_keys, exports
- View: user_credit_balance → { user_id, balance }

```typescript
import { createServerClient } from '@supabase/ssr'
import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'

export async function middleware(request: NextRequest) {
  let supabaseResponse = NextResponse.next({ request })

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return request.cookies.getAll()
        },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value }) =>
            request.cookies.set(name, value)
          )
          supabaseResponse = NextResponse.next({ request })
          cookiesToSet.forEach(({ name, value, options }) =>
            supabaseResponse.cookies.set(name, value, options)
          )
        },
      },
    }
  )

  // Refresh session — MUST call getUser() not getSession()
  const { data: { user } } = await supabase.auth.getUser()

  // Redirect unauthenticated users away from dashboard
  if (!user && request.nextUrl.pathname.startsWith('/dashboard')) {
    return NextResponse.redirect(new URL('/login', request.url))
  }

  // Redirect authenticated users away from auth pages
  if (
    user &&
    (request.nextUrl.pathname === '/login' ||
     request.nextUrl.pathname === '/signup')
  ) {
    return NextResponse.redirect(new URL('/dashboard/scrape', request.url))
  }

  return supabaseResponse
}

export const config = {
```

### 7b. The scrape invocation loop

This is the core function that runs when user submits the form.
It chains multiple edge function calls until done=true.

```typescript
const runScrapeJob = async ({
  videoUrl,
  maxComments,
  filters,
  onProgress,
  onComplete,
  onError,
}: {
  videoUrl:    string
  maxComments: number | null
  filters:     FetchFilters
  onProgress:  (jobId: string, commentCount: number, creditsUsed: number) => void
  onComplete:  (jobId: string, totalComments: number) => void
  onError:     (message: string) => void
}) => {
  const supabase = createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) { onError('Not authenticated'); return }

  // Get the session access token to pass to the edge function
  const { data: { session } } = await supabase.auth.getSession()
  if (!session) { onError('No session'); return }

  let jobId: string | null = null
  let pageToken: string | null = null
  let totalComments = 0
  let totalCreditsUsed = 0
  let invocationCount = 0
  const MAX_INVOCATIONS = 100 // safety cap: 100 × 2000 = 200,000 comments max

  while (invocationCount < MAX_INVOCATIONS) {
    invocationCount++

    // Build request body
    const body: Record<string, unknown> = {
      videoUrl,
      maxComments,
      filters,
    }
    // On 2nd+ calls, pass jobId and pageToken to resume
    if (jobId)    body.jobId     = jobId
    if (pageToken) body.pageToken = pageToken

    // Call the edge function directly via fetch
    // (supabase.functions.invoke also works but fetch gives better error handling)
    const response = await fetch(
      `${process.env.NEXT_PUBLIC_SUPABASE_URL}/functions/v1/fetch-comments`,
      {
        method:  'POST',
        headers: {
          'Content-Type':  'application/json',
          'Authorization': `Bearer ${session.access_token}`,
          'apikey':        process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
        },
        body: JSON.stringify(body),
      }
    )

    const result = await response.json()

    // Handle errors
    if (!response.ok || result.error) {
      if (result.error === 'insufficient_credits') {
        onError(`Not enough credits. You need more credits to continue.`)
        return
      }
      onError(result.error ?? result.message ?? 'Scraping failed')
      return
    }

    // Update state from this invocation's response
    jobId          = result.job_id
    pageToken      = result.nextPageToken
    totalComments  = result.comment_count   // accumulated total from edge function
    totalCreditsUsed += result.credits_used

    // Fire progress callback so UI updates
    onProgress(jobId, totalComments, totalCreditsUsed)

    // Stop if cancelled
    if (result.cancelled) {
      onError('Job was cancelled')
      return
    }

    // Stop if all pages fetched
    if (result.done) {
      onComplete(jobId, totalComments)
      return
    }

    // More pages remain — loop again with jobId + pageToken
    // Small delay to avoid hammering (edge function has its own rate limiting)
    await new Promise(resolve => setTimeout(resolve, 500))
  }

  onError('Too many invocations — job may be very large. Check your jobs page.')
}
```

```typescript
// lib/hooks/useComments.ts
'use client'
import { useState, useCallback } from 'react'
import { createClient } from '@/lib/supabase/client'

const PAGE_SIZE = 50

export function useComments(jobId: string | null) {
  const [comments, setComments] = useState<Comment[]>([])
  const [loading, setLoading] = useState(false)
  const [hasMore, setHasMore] = useState(true)
  const [offset, setOffset] = useState(0)
  const supabase = createClient()

  const loadInitial = useCallback(async () => {
    if (!jobId) return
    setLoading(true)
    setComments([])
    setOffset(0)
    const { data } = await supabase
      .from('comments')
      .select('*')
      .eq('job_id', jobId)
      .order('likes', { ascending: false })
      .range(0, PAGE_SIZE - 1)
    setComments(data ?? [])
    setHasMore((data?.length ?? 0) === PAGE_SIZE)
    setOffset(PAGE_SIZE)
    setLoading(false)
  }, [jobId])

  const loadMore = useCallback(async () => {
    if (!jobId || loading || !hasMore) return
    setLoading(true)
    const { data } = await supabase
      .from('comments')
      .select('*')
      .eq('job_id', jobId)
      .order('likes', { ascending: false })
      .range(offset, offset + PAGE_SIZE - 1)
    setComments(prev => [...prev, ...(data ?? [])])
    setHasMore((data?.length ?? 0) === PAGE_SIZE)
    setOffset(prev => prev + PAGE_SIZE)
    setLoading(false)
  }, [jobId, offset, loading, hasMore])

  return { comments, loading, hasMore, loadInitial, loadMore }
}
```

```typescript
import { createClient } from '@/lib/supabase/client'

export const fetchAllCommentsForExport = async (jobId: string) => {
  const supabase = createClient()
  const all: any[] = []
  let offset = 0
  while (true) {
    const { data } = await supabase
      .from('comments')
      .select('*')
      .eq('job_id', jobId)
      .order('likes', { ascending: false })
      .range(offset, offset + 999)

    if (data.length < 1000) break

export const downloadCSV = (comments: any[], filename: string) => {
  const headers = ['Published','Author','Author Channel','Comment','Likes','Reply Count','Is Reply','Parent ID']
  const esc = (v: unknown) => `"${String(v ?? '').replace(/"/g, '""')}"`
  const rows = comments.map(c => [
    c.published_at ? new Date(c.published_at).toISOString() : '',
    esc(c.author), esc(c.author_channel), esc(c.text),
    c.likes ?? 0, c.reply_count ?? 0,
    c.is_reply ? 'true' : 'false', c.parent_id ?? '',
  ].join(','))
  const csv = '\uFEFF' + [headers.join(','), ...rows].join('\n')
  const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' })
  const url = URL.createObjectURL(blob)
  const a = document.createElement('a')
  a.href = url; a.download = `${filename}_comments.csv`
  document.body.appendChild(a); a.click(); document.body.removeChild(a)
  URL.revokeObjectURL(url)
}
```

```typescript
// Fetch jobs for the logged-in user
const fetchJobs = async (page = 0) => {
  const supabase = createClient()
  const PAGE_SIZE = 20
  const { data, count } = await supabase
    .from('jobs')
    .select('*', { count: 'exact' })
    .order('created_at', { ascending: false })
    .range(page * PAGE_SIZE, (page + 1) * PAGE_SIZE - 1)
  return { jobs: data ?? [], total: count ?? 0 }
}
// Note: RLS automatically filters to the logged-in user's jobs only
// Do NOT add .eq('user_id', ...) — RLS handles it

// Stats for the stats cards:
const fetchJobStats = async () => {
  const supabase = createClient()
  // Total jobs
  const { count: totalJobs } = await supabase
    .from('jobs').select('*', { count: 'exact', head: true })

  // Total comments scraped
  const { data: commentStats } = await supabase
    .from('jobs')
    .select('downloaded_comments')
    .eq('status', 'completed')

  const totalComments = commentStats?.reduce((sum, j) => sum + (j.downloaded_comments ?? 0), 0) ?? 0

  // Credits spent (absolute value of negative ledger entries)
  const { data: creditData } = await supabase
    .from('credit_ledger')
    .select('amount')
    .lt('amount', 0)

  const creditsSpent = Math.abs(
    creditData?.reduce((sum, r) => sum + r.amount, 0) ?? 0
  )

  return { totalJobs, totalComments, creditsSpent }
}

// Retry a failed job:
const retryJob = async (job: Job) => {
  const supabase = createClient()
  const { data: { session } } = await supabase.auth.getSession()
  if (!session) return

  // Reset job status
  await supabase.from('jobs').update({
    status: 'queued',
    error_message: null,
    retry_count: (job.retry_count ?? 0) + 1,
    progress: 0,
    downloaded_comments: 0,
    started_at: null,
    completed_at: null,
  }).eq('id', job.id)

  // Re-invoke edge function (fresh call, same job ID)
  await fetch(
    `${process.env.NEXT_PUBLIC_SUPABASE_URL}/functions/v1/fetch-comments`,
    {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${session.access_token}`,
        'apikey': process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
      },
      body: JSON.stringify({
        videoUrl: job.video_url,
        maxComments: job.requested_comments,
        filters: job.filters ?? {},
        jobId: job.id,
      }),
    }
  )
  toast.success('Job restarted!')
}
```
