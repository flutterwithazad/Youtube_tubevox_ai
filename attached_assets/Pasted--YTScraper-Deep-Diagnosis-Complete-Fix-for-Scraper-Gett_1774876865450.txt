## ARCHITECTURE OVERVIEW (so you understand what to fix)

The scraper has 3 layers that can all fail independently:

1. SUPABASE EDGE FUNCTION (fetch-comments) — Deno runtime, 150s timeout
   Called repeatedly by the frontend in a loop until done=true

2. FRONTEND LOOP (Next.js) — chains edge function calls in browser memory
   Dies if tab is closed, JWT expires, or any single call throws

3. DATABASE (Supabase Postgres) — jobs table, comments, credit_ledger
   Jobs can get stuck in 'running' with no recovery mechanism

```sql
CREATE OR REPLACE FUNCTION public.atomic_credit_deduct(
  p_user_id     UUID,
  p_batch_size  INT,
  p_source_id   UUID,
  p_description TEXT
)
RETURNS JSON
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_balance     BIGINT;
  v_new_balance BIGINT;
BEGIN
  -- Per-user advisory lock prevents concurrent double-spending
  PERFORM pg_advisory_xact_lock(hashtext(p_user_id::TEXT));

  -- Get real-time balance from ledger
  SELECT COALESCE(SUM(amount), 0)
  INTO v_balance
  FROM public.credit_ledger
  WHERE user_id = p_user_id;

  -- Insufficient credits
  IF v_balance < p_batch_size THEN
    RETURN json_build_object(
      'ok',      false,
      'balance', v_balance,
      'needed',  p_batch_size
    );
  END IF;

  -- Deduct
  v_new_balance := v_balance - p_batch_size;

  INSERT INTO public.credit_ledger (
    user_id, amount, source_type, source_id, description, balance_after
  ) VALUES (
    p_user_id,
    -p_batch_size,
    'job_reserve',
    p_source_id,
    p_description,
    v_new_balance
  );

  RETURN json_build_object(
    'ok',       true,
    'balance',  v_new_balance,
    'deducted', p_batch_size
  );
END;
$$;

GRANT EXECUTE ON FUNCTION public.atomic_credit_deduct(UUID, INT, UUID, TEXT)
  TO authenticated, service_role;
```

### FIX 4 — Edge function: add retry logic for YouTube API transient errors

In `youtubeApi.ts`, the `ytGet` method already rotates keys on quota errors.
Add retry logic for transient network errors and 5xx errors:

```typescript
private async ytGet(baseUrl: string): Promise<YTResponse> {
  let attempts = 0
  const MAX_ATTEMPTS = 10
  const RETRY_DELAYS = [1000, 2000, 4000, 8000] // exponential backoff ms

  while (attempts < MAX_ATTEMPTS) {
    attempts++
    try {
      const k   = await this.key()
      const url = `${baseUrl}&key=${k.key_value}`
      const res = await fetch(url)
      const data: YTResponse = await res.json()

      if (!data.error) {
        await this.km.markUsed(k.id, 1)
        return data
      }

      const { code, message } = data.error

      // Quota exhausted → rotate key
      if (code === 403 && message.toLowerCase().includes('quota')) {
        console.log(`[KEY_ROTATE] Key "${k.label}" quota exceeded`)
        this.activeKey = await this.km.rotateKey(k.id)
        continue // retry immediately with new key
      }

      // Comments disabled on this video — unrecoverable
      if (code === 403) throw new Error('Comments are disabled on this video')

      // Video not found / private — unrecoverable
      if (code === 404) throw new Error('Video not found or is private')

      // Rate limit (429) → wait then retry
      if (code === 429 || res.status === 429) {
        const delay = RETRY_DELAYS[Math.min(attempts - 1, RETRY_DELAYS.length - 1)]
        console.log(`[RATE_LIMIT] Waiting ${delay}ms before retry ${attempts}`)
        await new Promise(r => setTimeout(r, delay))
        continue
      }

      // 5xx server errors → wait then retry
      if (code >= 500 || res.status >= 500) {
        const delay = RETRY_DELAYS[Math.min(attempts - 1, RETRY_DELAYS.length - 1)]
        console.log(`[SERVER_ERROR] HTTP ${code} — retrying in ${delay}ms`)
        await new Promise(r => setTimeout(r, delay))
        continue
      }

      // Any other error — mark key and throw
      await this.km.markError(k.id)
      throw new Error(message)

    } catch (err: any) {
      // Network errors (fetch failed) → retry with backoff
      if (err.message?.includes('fetch') || err.name === 'TypeError') {
        const delay = RETRY_DELAYS[Math.min(attempts - 1, RETRY_DELAYS.length - 1)]
        console.log(`[NETWORK_ERROR] ${err.message} — retrying in ${delay}ms`)
        await new Promise(r => setTimeout(r, delay))
        continue
      }
      throw err // unrecoverable error
    }
  }

  throw new Error('Max retry attempts reached — YouTube API unavailable')
}
```

### FIX 7 — Frontend: add retry with backoff for failed invocations

Don't let a single network error kill the entire scrape.
Add retry logic in the fetch loop:

```typescript
const callEdgeFunction = async (body: object, retries = 3): Promise<any> => {
  for (let attempt = 1; attempt <= retries; attempt++) {
    try {
      const { data: { session } } = await supabase.auth.getSession()
      if (!session) throw new Error('No session')

      const response = await fetch(
        `${process.env.NEXT_PUBLIC_SUPABASE_URL}/functions/v1/fetch-comments`,
        {
          method:  'POST',
          headers: {
            'Content-Type':  'application/json',
            'Authorization': `Bearer ${session.access_token}`,
            'apikey':        process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
          },
          body: JSON.stringify(body),
          signal: AbortSignal.timeout(165_000), // 165s — slightly over edge fn limit
        }
      )

      if (!response.ok) {
        const errorText = await response.text()
        throw new Error(`HTTP ${response.status}: ${errorText}`)
      }

      return await response.json()

    } catch (err: any) {
      const isLastAttempt = attempt === retries
      const isTimeout = err.name === 'TimeoutError' || err.name === 'AbortError'
      const isNetwork = err.message?.includes('fetch') || err.name === 'TypeError'

      if (!isLastAttempt && (isTimeout || isNetwork)) {
        const delay = attempt * 2000 // 2s, 4s, 6s
        console.warn(`[RETRY] Attempt ${attempt} failed: ${err.message} — retrying in ${delay}ms`)
        await new Promise(r => setTimeout(r, delay))
        continue
      }

      throw err // give up after max retries
    }
  }
}
```

### FIX 9 — Database: add watchdog function to auto-recover stuck jobs

Run in Supabase SQL Editor to create a watchdog that marks
stuck jobs as failed and refunds credits:

```sql
CREATE OR REPLACE FUNCTION public.recover_stuck_jobs()
RETURNS INTEGER
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
DECLARE
  v_count INTEGER := 0;
  v_job   RECORD;
BEGIN
  -- Find jobs stuck in 'running' for more than 15 minutes
  FOR v_job IN
    SELECT id, user_id, credits_reserved, downloaded_comments
    FROM public.jobs
    WHERE status = 'running'
      AND started_at < NOW() - INTERVAL '15 minutes'
  LOOP
    -- Mark as failed
    UPDATE public.jobs
    SET
      status        = 'failed',
      completed_at  = NOW(),
      error_message = 'Auto-recovered: job exceeded maximum run time. Credits refunded.'
    WHERE id = v_job.id;

    -- Refund any unreturned credits
    IF v_job.credits_reserved > 0 THEN
      INSERT INTO public.credit_ledger (
        user_id, amount, source_type, source_id, description
      ) VALUES (
        v_job.user_id,
        v_job.credits_reserved,
        'job_cancel_refund',
        v_job.id,
        'Auto-refund: job timed out after 15 minutes'
      );
    END IF;

    v_count := v_count + 1;
  END LOOP;

  RETURN v_count;
END;
$$;

GRANT EXECUTE ON FUNCTION public.recover_stuck_jobs() TO service_role;
```
