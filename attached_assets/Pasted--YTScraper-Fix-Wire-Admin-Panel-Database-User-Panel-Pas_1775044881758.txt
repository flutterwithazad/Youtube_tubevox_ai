## UNDERSTAND THE FULL CHAIN

For maintenance mode to work, ALL of these must be true:

  [Admin toggle click]
       ↓
  [Admin panel saves to DB]        ← platform_settings WHERE key='maintenance_mode'
       ↓
  [DB value = 'true']              ← verify this in Supabase SQL Editor
       ↓
  [Next.js middleware reads DB]    ← runs on every user request
       ↓
  [Middleware redirects to /maintenance]
       ↓
  [User sees maintenance page]

If ANY step in this chain is broken, the feature doesn't work.
We will test and fix each step individually.

### B1: Create the admin API route to save settings

This project uses Express.js for the admin backend.
Find the admin API routes file (likely `server/routes/admin.js` or similar).
Add these two routes:

```javascript
// GET /api/admin/settings — load all platform settings
router.get('/settings', requireAdmin, async (req, res) => {
  try {
    const { data, error } = await supabaseAdmin
      .from('platform_settings')
      .select('key, value, description, updated_at')
      .order('key')

    if (error) throw error
    res.json({ settings: data })
  } catch (err) {
    console.error('[ADMIN] GET /settings error:', err.message)
    res.status(500).json({ error: err.message })
  }
})

// PATCH /api/admin/settings/:key — update one setting
router.patch('/settings/:key', requireAdmin, async (req, res) => {
  const { key }   = req.params
  const { value } = req.body

  // Whitelist allowed keys
  const ALLOWED_KEYS = [
    'maintenance_mode',
    'new_signups_enabled',
    'free_plan_credits',
  ]
  if (!ALLOWED_KEYS.includes(key)) {
    return res.status(400).json({ error: `Unknown setting key: ${key}` })
  }

  // Validate values per key
  if (key === 'maintenance_mode' || key === 'new_signups_enabled') {
    if (value !== 'true' && value !== 'false') {
      return res.status(400).json({ error: 'Value must be "true" or "false"' })
    }
  }

  try {
    const { error } = await supabaseAdmin
      .from('platform_settings')
      .upsert(
        {
          key,
          value: String(value),
          updated_at: new Date().toISOString(),
          updated_by: req.admin?.id ?? null,
        },
        { onConflict: 'key' }
      )

    if (error) throw error

    console.log(`[ADMIN] Setting updated: ${key} = ${value} by admin ${req.admin?.email}`)
    res.json({ success: true, key, value })
  } catch (err) {
    console.error('[ADMIN] PATCH /settings/:key error:', err.message)
    res.status(500).json({ error: err.message })
  }
})
```

```typescript
// State for both toggles
const [maintenanceMode, setMaintenanceMode] = useState(false)

const [saving, setSaving] = useState<string | null>(null)

// Load current values when page mounts

  loadSettings()
}, [])

const loadSettings = async () => {
  try {
    const res = await fetch('/api/admin/settings', {
      credentials: 'include',
    })
    if (!res.ok) throw new Error('Failed to load settings')
    const { settings } = await res.json()

    const maintenance = settings.find((s: any) => s.key === 'maintenance_mode')
    const signups     = settings.find((s: any) => s.key === 'new_signups_enabled')

    setMaintenanceMode(maintenance?.value === 'true')
    setSignupsEnabled(signups?.value !== 'false') // default true if missing
  } catch (err) {
    console.error('Failed to load settings:', err)
    toast.error('Failed to load platform settings')
  }
}

const updateSetting = async (key: string, value: string): Promise<boolean> => {
  setSaving(key)
  try {
    const res = await fetch(`/api/admin/settings/${key}`, {
      method:      'PATCH',
      credentials: 'include',
      headers:     { 'Content-Type': 'application/json' },
      body:        JSON.stringify({ value }),
    })
    const data = await res.json()
    if (!res.ok) throw new Error(data.error ?? 'Save failed')
    return true
  } catch (err: any) {
    toast.error(`Failed to save setting: ${err.message}`)
    return false
  } finally {
    setSaving(null)
  }
}

// Maintenance mode toggle handler
const handleMaintenanceToggle = async (checked: boolean) => {
  // Optimistic update
  setMaintenanceMode(checked)

  const saved = await updateSetting('maintenance_mode', checked ? 'true' : 'false')

  if (saved) {
    toast.success(
      checked
        ? '⚠ Maintenance mode is now ON — users cannot access the app'
        : '✓ Maintenance mode is OFF — app is back online',
      { duration: 5000 }
    )
  } else {
    // Revert optimistic update if save failed
    setMaintenanceMode(!checked)
  }
}

// Signups toggle handler
const handleSignupsToggle = async (checked: boolean) => {
  setSignupsEnabled(checked)

  const saved = await updateSetting('new_signups_enabled', checked ? 'true' : 'false')

  if (saved) {
    toast.success(
      checked
        ? '✓ New signups are now open'
        : '⚠ New signups are now CLOSED — registration blocked'
    )
  } else {
    setSignupsEnabled(!checked)
  }
}
```

## FIX C — User panel middleware: read maintenance mode from DB correctly

This is where the enforcement happens. The Next.js middleware in the
USER PANEL (not admin panel) must read from `platform_settings` on
every request and redirect to /maintenance if value is 'true'.

Find `middleware.ts` in the user panel Next.js app.
Replace it entirely with this exact code:

```typescript
import { createServerClient } from '@supabase/ssr'
import { NextResponse }       from 'next/server'
import type { NextRequest }   from 'next/server'

export async function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl

  // These paths are never blocked by maintenance mode or auth checks
  const ALWAYS_ALLOW = [
    '/maintenance',
    '/auth',
    '/_next',
    '/favicon.ico',
    '/api/health',
  ]

  const isAlwaysAllowed = ALWAYS_ALLOW.some(p => pathname.startsWith(p))
  const isAdminRoute    = pathname.startsWith('/admin')
  const isStaticFile    = /\.(?:svg|png|jpg|jpeg|gif|webp|ico|css|js|woff|woff2)$/.test(pathname)

  if (isStaticFile) return NextResponse.next({ request })

  // Build response object we'll mutate for cookie handling
  let response = NextResponse.next({ request })

  // Create Supabase client with cookie handling
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return request.cookies.getAll()
        },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value }) =>
            request.cookies.set(name, value)
          )
          response = NextResponse.next({ request })
          cookiesToSet.forEach(({ name, value, options }) =>
            response.cookies.set(name, value, options)
          )
        },
      },
    }
  )

  // ════════════════════════════════════════════════════
  // CHECK 1: MAINTENANCE MODE
  // Applies to: all routes EXCEPT admin, /maintenance, static files
  // ════════════════════════════════════════════════════
  if (!isAdminRoute && !isAlwaysAllowed) {
    try {
      const { data: setting, error } = await supabase
        .from('platform_settings')
        .select('value')
        .eq('key', 'maintenance_mode')
        .single()

      console.log(
        `[MW] ${pathname} | maintenance_mode=${setting?.value} | error=${error?.message ?? 'none'}`
      )

      if (setting?.value === 'true') {
        console.log(`[MW] Redirecting ${pathname} → /maintenance`)
        return NextResponse.redirect(new URL('/maintenance', request.url))
      }
    } catch (err: any) {
      // If DB is unreachable, let users through (fail open) — don't block everyone
      console.error('[MW] Failed to check maintenance_mode:', err.message)
    }
  }

  // ════════════════════════════════════════════════════
  // CHECK 2: AUTH — get current user
  // ════════════════════════════════════════════════════
  const {
    data: { user },
    error: authError,
  } = await supabase.auth.getUser()

  if (authError) {
    console.log('[MW] Auth error:', authError.message)
  }

  // Protect /dashboard routes — redirect to login if not authenticated
  if (!user && pathname.startsWith('/dashboard')) {
    const loginUrl = new URL('/login', request.url)
    loginUrl.searchParams.set('next', pathname)
    return NextResponse.redirect(loginUrl)
  }

  // Redirect logged-in users away from login/signup pages
  if (user && (pathname === '/login' || pathname === '/signup')) {
    return NextResponse.redirect(new URL('/dashboard/scrape', request.url))
  }

  // ════════════════════════════════════════════════════
  // CHECK 3: SUSPENSION — check profile on every dashboard request
  // ════════════════════════════════════════════════════
  if (user && pathname.startsWith('/dashboard')) {
    try {
      const { data: profile } = await supabase
        .from('profiles')
        .select('is_suspended, suspended_reason, account_status')
        .eq('id', user.id)
        .single()

      console.log(
        `[MW] User ${user.id} | is_suspended=${profile?.is_suspended} | status=${profile?.account_status}`
      )

      const isSuspended =
        profile?.is_suspended === true ||
        profile?.account_status === 'suspended'

      if (isSuspended) {
        // Sign out
        await supabase.auth.signOut()

        // Redirect to login with reason
        const loginUrl = new URL('/login', request.url)
        loginUrl.searchParams.set('suspended', 'true')
        if (profile?.suspended_reason) {
          loginUrl.searchParams.set(
            'reason',
            encodeURIComponent(profile.suspended_reason)
          )
        }
        console.log(`[MW] Suspended user ${user.id} → redirecting to login`)
        return NextResponse.redirect(loginUrl)
      }
    } catch (err: any) {
      console.error('[MW] Failed to check suspension:', err.message)
    }
  }

  // ════════════════════════════════════════════════════
  // CHECK 4: NEW SIGNUPS — block /signup if disabled
  // ════════════════════════════════════════════════════
  if (pathname === '/signup') {
    try {
      const { data: setting } = await supabase
        .from('platform_settings')
        .select('value')
        .eq('key', 'new_signups_enabled')
        .single()

      if (setting?.value === 'false') {
        console.log('[MW] Signups disabled → redirecting /signup → /registration-closed')
        return NextResponse.redirect(new URL('/registration-closed', request.url))
      }
    } catch {
      // fail open
    }
  }

  return response
}

export const config = {
  matcher: [
    /*
     * Match ALL paths except:
     * - _next/static (static files)
     * - _next/image (image optimization)
     * - favicon.ico
     * - image files
     */
    '/((?!_next/static|_next/image|favicon\\.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)',
  ],
}
```

## COMMON MISTAKES TO CHECK

1. **Middleware is in admin panel instead of user panel** — maintenance_mode must be
   enforced in the USER PANEL Next.js middleware.ts, not the admin panel.

2. **Middleware matcher is too narrow** — if matcher only covers `/dashboard/*`,
   then visiting `/` or `/login` won't be blocked. Use the broad matcher above.

3. **Admin panel uses different base URL** — if admin panel calls
   `/api/admin/settings` but the route is actually `/admin/api/settings`,
   the save fails silently. Check the exact route paths.

4. **Supabase environment variables missing in user panel** — middleware needs
   `NEXT_PUBLIC_SUPABASE_URL` and `NEXT_PUBLIC_SUPABASE_ANON_KEY`.
   Check these are set in Replit Secrets for the user panel.

5. **The maintenance page itself triggers middleware** — the `/maintenance` path
   must be in the ALWAYS_ALLOW list, otherwise it creates an infinite redirect loop.
