## WHAT WE ARE BUILDING

A user dashboard for a YouTube Comment Scraper SaaS. Users:
1. Paste a YouTube URL → choose comment count → click Scrape
2. Watch live progress (Edge Function does the scraping)
3. Browse all comments in a Comment Explorer (scrollable table with search)
4. Export as CSV / Excel / JSON with one click
5. Buy credit packs (payment is "coming soon" modal in MVP)
6. View credit balance and history

Business model: Credit packages only. NO subscriptions.
- 1 credit = 1,000 comments
- Signup bonus credits = dynamic (read from admin-controlled platform_settings table, NEVER hardcode)
- One job at a time per user

## SUPABASE EDGE FUNCTION

Create: supabase/functions/scrape-comments/index.ts

This Deno function handles ALL YouTube scraping server-side.
The frontend ONLY creates the job row and calls this function.

```typescript
import { serve } from 'https://deno.land/std@0.168.0/http/server.ts'
import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'

const cors = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
}

serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: cors })
  try {
    const supabaseAdmin = createClient(
      Deno.env.get('SUPABASE_URL')!,
      Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!
    )
    const authHeader = req.headers.get('Authorization')!
    const supabaseUser = createClient(
      Deno.env.get('SUPABASE_URL')!,
      Deno.env.get('SUPABASE_ANON_KEY')!,
      { global: { headers: { Authorization: authHeader } } }
    )
    const { data: { user }, error: authErr } = await supabaseUser.auth.getUser()
    if (authErr || !user) {
      return new Response(JSON.stringify({ error: 'Unauthorized' }), { status: 401, headers: { ...cors, 'Content-Type': 'application/json' } })
    }

    const { job_id } = await req.json()
    const { data: job } = await supabaseAdmin.from('jobs').select('*').eq('id', job_id).eq('user_id', user.id).single()
    if (!job) return new Response(JSON.stringify({ error: 'Job not found' }), { status: 404, headers: { ...cors, 'Content-Type': 'application/json' } })

    // Pick least-used API key from pool
    const { data: apiKey } = await supabaseAdmin.from('youtube_api_keys').select('*').eq('is_active', true).eq('is_paused', false).lt('consecutive_errors', 5).order('quota_used', { ascending: true }).limit(1).single()
    if (!apiKey) {
      await supabaseAdmin.from('jobs').update({ status: 'failed', error_message: 'No YouTube API keys available.', completed_at: new Date().toISOString() }).eq('id', job_id)
      await supabaseAdmin.from('credit_ledger').insert({ user_id: user.id, amount: +job.credits_reserved, source_type: 'job_cancel_refund', source_id: job_id, description: 'Refund: no API keys available' })
      return new Response(JSON.stringify({ error: 'No API keys' }), { status: 503, headers: { ...cors, 'Content-Type': 'application/json' } })
    }

    await supabaseAdmin.from('jobs').update({ status: 'running', started_at: new Date().toISOString() }).eq('id', job_id)

    const YT = apiKey.key_value
    const videoId = job.video_id
    const maxFetch = job.requested_comments
    let pageToken: string | null = null
    let totalFetched = 0
    let videoTitle = job.video_title
    let channelName = job.channel_name

    // Fetch video metadata
    if (!videoTitle) {
      const vRes = await fetch(`https://www.googleapis.com/youtube/v3/videos?part=snippet,statistics&id=${videoId}&key=${YT}`)
      const vData = await vRes.json()
      if (vData.items?.[0]) {
        const s = vData.items[0].snippet, st = vData.items[0].statistics
        videoTitle = s.title; channelName = s.channelTitle
        await supabaseAdmin.from('jobs').update({ video_title: videoTitle, channel_name: channelName, thumbnail: s.thumbnails?.medium?.url, comment_count: parseInt(st.commentCount ?? '0'), view_count: parseInt(st.viewCount ?? '0'), like_count: parseInt(st.likeCount ?? '0') }).eq('id', job_id)
      }
    }

    // Scrape comment pages
    while (totalFetched < maxFetch) {
      const batchSize = Math.min(100, maxFetch - totalFetched)
      let url = `https://www.googleapis.com/youtube/v3/commentThreads?part=snippet&videoId=${videoId}&maxResults=${batchSize}&textFormat=plainText&key=${YT}`
      if (pageToken) url += `&pageToken=${pageToken}`
      const res = await fetch(url)
      const data = await res.json()
      if (data.error) {
        await supabaseAdmin.from('youtube_api_keys').update({ error_count: apiKey.error_count + 1, consecutive_errors: apiKey.consecutive_errors + 1, last_error_at: new Date().toISOString(), last_error_msg: data.error.message }).eq('id', apiKey.id)
        throw new Error(`YouTube API: ${data.error.message}`)
      }
      const items = data.items ?? []
      const toInsert: Record<string, unknown>[] = []
      for (const item of items) {
        const s = item.snippet.topLevelComment.snippet
        toInsert.push({ job_id, comment_id: item.id, author: s.authorDisplayName, author_channel: s.authorChannelUrl, author_channel_id: s.authorChannelId?.value, text: s.textDisplay, text_original: s.textOriginal, likes: s.likeCount ?? 0, reply_count: item.snippet.totalReplyCount ?? 0, is_reply: false, published_at: s.publishedAt })
        if (item.snippet.totalReplyCount > 0 && job.filters?.include_replies === true) {
          const rRes = await fetch(`https://www.googleapis.com/youtube/v3/comments?part=snippet&parentId=${item.id}&maxResults=100&key=${YT}`)
          const rData = await rRes.json()
          for (const r of rData.items ?? []) {
            toInsert.push({ job_id, comment_id: r.id, author: r.snippet.authorDisplayName, author_channel: r.snippet.authorChannelUrl, text: r.snippet.textDisplay, likes: r.snippet.likeCount ?? 0, reply_count: 0, is_reply: true, parent_id: item.id, published_at: r.snippet.publishedAt })
          }
        }
      }
      if (toInsert.length > 0) await supabaseAdmin.from('comments').upsert(toInsert, { onConflict: 'job_id,comment_id', ignoreDuplicates: true })
      totalFetched += items.length
      pageToken = data.nextPageToken ?? null
      const progress = Math.min(100, Math.round((totalFetched / maxFetch) * 100))
      await supabaseAdmin.from('jobs').update({ downloaded_comments: totalFetched, progress, next_page_token: pageToken }).eq('id', job_id)
      await supabaseAdmin.from('youtube_api_keys').update({ quota_used: apiKey.quota_used + 1, last_used_at: new Date().toISOString(), consecutive_errors: 0 }).eq('id', apiKey.id)
      if (!pageToken) break
    }

    // Settle credits
    const creditsUsed = Math.max(1, Math.ceil(totalFetched / 1000))
    const refund = job.credits_reserved - creditsUsed
    if (refund > 0) await supabaseAdmin.from('credit_ledger').insert({ user_id: user.id, amount: +refund, source_type: 'job_settle', source_id: job_id, description: `Refund: used ${creditsUsed} of ${job.credits_reserved} reserved` })
    await supabaseAdmin.from('jobs').update({ status: 'completed', progress: 100, credits_used: creditsUsed, downloaded_comments: totalFetched, completed_at: new Date().toISOString() }).eq('id', job_id)
    await supabaseAdmin.from('notifications').insert({ user_id: user.id, type: 'job_complete', title: 'Scrape complete!', body: `Downloaded ${totalFetched.toLocaleString()} comments from "${videoTitle}"`, action_url: `/dashboard/jobs/${job_id}`, channel: 'in_app' })
    await supabaseAdmin.from('usage_logs').insert({ user_id: user.id, job_id, credits_used: creditsUsed, action: 'job_complete', metadata: { downloaded: totalFetched } })

    return new Response(JSON.stringify({ success: true, downloaded: totalFetched, credits_used: creditsUsed }), { headers: { ...cors, 'Content-Type': 'application/json' } })
  } catch (err) {
    console.error(err)
    return new Response(JSON.stringify({ error: (err as Error).message }), { status: 500, headers: { ...cors, 'Content-Type': 'application/json' } })
  }
})
```

## MIDDLEWARE (middleware.ts)

```typescript
import { createServerClient } from '@supabase/ssr'
import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'
export async function middleware(request: NextRequest) {
  let supabaseResponse = NextResponse.next({ request })
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    { cookies: {
        getAll() { return request.cookies.getAll() },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value }) => request.cookies.set(name, value))
          supabaseResponse = NextResponse.next({ request })
          cookiesToSet.forEach(({ name, value, options }) => supabaseResponse.cookies.set(name, value, options))
        },
    }}
  )
  const { data: { user } } = await supabase.auth.getUser()
  if (!user && request.nextUrl.pathname.startsWith('/dashboard')) {
    return NextResponse.redirect(new URL('/login', request.url))
  }
  if (user && ['/login', '/signup'].includes(request.nextUrl.pathname)) {
    return NextResponse.redirect(new URL('/dashboard/scrape', request.url))
  }
  return supabaseResponse
}
export const config = { matcher: ['/dashboard/:path*', '/login', '/signup'] }
```

const fetchAllForExport = async (jobId: string) => {
  const all = []
  let offset = 0
  const id = toast.loading('Preparing export...')
  while (true) {
    const { data } = await supabase.from('comments').select('*')
      .eq('job_id', jobId).order('likes', { ascending: false })
      .range(offset, offset + 999)
    if (!data || data.length === 0) break
    all.push(...data)
    if (data.length < 1000) break
    offset += 1000
  }
  toast.dismiss(id)
  return all
}

  const escape = (v: string) => `"${String(v ?? '').replace(/"/g, '""')}"`
  const rows = comments.map(c => [
    c.published_at ? new Date(c.published_at).toISOString() : '',
    escape(c.author ?? ''), escape(c.author_channel ?? ''), escape(c.text ?? ''),
    c.likes ?? 0, c.reply_count ?? 0, c.is_reply ? 'true' : 'false',
    c.parent_id ?? '', c.language ?? '',
  ].join(','))
  const csv = '\uFEFF' + [headers.join(','), ...rows].join('\n')
  const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' })
  const url = URL.createObjectURL(blob)
  const a = document.createElement('a')
  a.href = url; a.download = `${filename}_comments.csv`
  document.body.appendChild(a); a.click(); document.body.removeChild(a)
  URL.revokeObjectURL(url)
  toast.success(`✓ Downloaded ${comments.length.toLocaleString()} comments as CSV`)
}

## CRITICAL RULES

1. NEVER hardcode credit amounts. Always read 'free_plan_credits' from platform_settings.
2. Credit balance = ALWAYS from user_credit_balance VIEW. Never compute SUM manually.
3. ONE JOB AT A TIME. Check for active jobs before creating new one.
4. Edge Function handles ALL scraping. Frontend only creates job row and invokes function.
5. CommentExplorer is ONE reusable component. Used in scrape page AND jobs/[id] page.
6. All exports are 100% client-side. No server upload for MVP. Direct browser download.
7. All Supabase frontend calls use anon key. RLS policies handle security automatically.
8. Skeleton loaders everywhere. No blank screens while data loads.
9. Wrap all Supabase calls in try/catch. Show sonner toast on errors.
10. If job fails: automatically refund credits_reserved back to user.
