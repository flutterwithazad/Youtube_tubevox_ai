### Scroll Animations
Add a scroll reveal effect to these sections as they enter viewport:

- Testimonial cards: fade up with 80ms stagger

- Pricing cards: scale up from 0.95 to 1.0

Use a simple `useInView` hook implemented with IntersectionObserver — no libraries needed:
```typescript
// hooks/useInView.ts
import { useEffect, useRef, useState } from 'react'
export function useInView(threshold = 0.1) {
  const ref = useRef<HTMLDivElement>(null)
  const [inView, setInView] = useState(false)
  useEffect(() => {
    const observer = new IntersectionObserver(
      ([entry]) => { if (entry.isIntersecting) setInView(true) },
      { threshold }
    )
    if (ref.current) observer.observe(ref.current)
    return () => observer.disconnect()
  }, [])
  return { ref, inView }
}
```
