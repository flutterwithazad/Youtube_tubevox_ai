## FEATURE 1 — MAINTENANCE MODE

### How it works:
- Admin toggles "Maintenance mode" ON in admin panel
- `platform_settings` row: key='maintenance_mode', value='true'
- Every user hitting ANY dashboard route sees a maintenance page instead
- Admin panel itself is NOT affected (admins can still work)
- When admin turns it OFF → users immediately get access back

---

### STEP 1C — Next.js middleware: enforce maintenance mode on every request

This is the most important part. The middleware runs on EVERY request
before the page renders, so it catches users instantly.

Update `middleware.ts`:

```typescript
import { createServerClient } from '@supabase/ssr'
import { NextResponse }       from 'next/server'
import type { NextRequest }   from 'next/server'

export async function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl

  // Admin panel is NEVER blocked by maintenance mode
  const isAdminRoute  = pathname.startsWith('/admin')
  const isMaintenancePage = pathname === '/maintenance'
  const isAuthCallback    = pathname.startsWith('/auth')

  let supabaseResponse = NextResponse.next({ request })

  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll()            { return request.cookies.getAll() },
        setAll(cookiesToSet) {
          cookiesToSet.forEach(({ name, value }) => request.cookies.set(name, value))
          supabaseResponse = NextResponse.next({ request })
          cookiesToSet.forEach(({ name, value, options }) =>
            supabaseResponse.cookies.set(name, value, options)
          )
        },
      },
    }
  )

  // ── Check maintenance mode (skip for admin + maintenance page itself) ──
  if (!isAdminRoute && !isMaintenancePage && !isAuthCallback) {
    try {
      const { data: setting } = await supabase
        .from('platform_settings')
        .select('value')
        .eq('key', 'maintenance_mode')
        .single()

      if (setting?.value === 'true') {
        // Redirect everyone (logged in or not) to maintenance page
        const maintenanceUrl = new URL('/maintenance', request.url)
        return NextResponse.redirect(maintenanceUrl)
      }
    } catch {
      // If DB read fails, don't block — fail open
    }
  }

  // ── Auth session check ────────────────────────────────────────
  const { data: { user } } = await supabase.auth.getUser()

  // Protect dashboard routes
  if (!user && pathname.startsWith('/dashboard')) {
    return NextResponse.redirect(new URL('/login', request.url))
  }

  // Redirect logged-in users away from auth pages
  if (user && (pathname === '/login' || pathname === '/signup')) {
    return NextResponse.redirect(new URL('/dashboard/scrape', request.url))
  }

  // ── Check if suspended user is trying to access dashboard ────
  if (user && pathname.startsWith('/dashboard')) {
    try {
      const { data: profile } = await supabase
        .from('profiles')
        .select('is_suspended, suspended_reason, account_status')
        .eq('id', user.id)
        .single()

      if (profile?.is_suspended || profile?.account_status === 'suspended') {
        // Sign them out server-side
        await supabase.auth.signOut()

        // Redirect to login with suspension reason in query params
        const loginUrl = new URL('/login', request.url)
        loginUrl.searchParams.set('suspended', 'true')
        if (profile?.suspended_reason) {
          loginUrl.searchParams.set('reason', profile.suspended_reason)
        }
        return NextResponse.redirect(loginUrl)
      }
    } catch {
      // If profile check fails, don't block
    }
  }

  return supabaseResponse
}

export const config = {
  matcher: [
    '/((?!_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)',
  ],
}
```

## FEATURE 2 — USER SUSPENSION WITH IMMEDIATE LOGOUT

### How it works:
- Admin suspends a user with a reason in the admin panel
- DB: `profiles.is_suspended = true`, `profiles.suspended_reason = 'reason text'`
- If user is currently logged in: they are immediately redirected out on next request
- The middleware (already updated above) checks `is_suspended` on every dashboard request
- On login attempt: show suspension message with the reason
- User cannot log in again until unsuspended

---

### STEP 2A — Admin panel: suspension action (already has UI)

Wire the existing suspend button to actually call the API:

```typescript
// In the user detail page or users table:
const suspendUser = async (userId: string, reason: string) => {
  if (!reason.trim()) {
    toast.error('Please enter a suspension reason')
    return
  }

  const response = await fetch(`/api/admin/users/${userId}/suspend`, {
    method:      'POST',
    credentials: 'include',
    headers:     { 'Content-Type': 'application/json' },
    body:        JSON.stringify({ reason }),
  })

  if (!response.ok) {
    toast.error('Failed to suspend user')
    return
  }

  toast.success('User suspended. They will be logged out immediately on next request.')
  // Refresh user data
  await fetchUser(userId)
}

const unsuspendUser = async (userId: string) => {
  const response = await fetch(`/api/admin/users/${userId}/unsuspend`, {
    method:      'POST',
    credentials: 'include',
  })

  if (!response.ok) {
    toast.error('Failed to unsuspend user')
    return
  }

  toast.success('User unsuspended. They can log in again.')
  await fetchUser(userId)
}
```

### STEP 2B — Admin API route: suspend endpoint

In `/api/admin/users/[userId]/suspend/route.ts`:

```typescript
import { createSupabaseAdmin } from '@/lib/supabase-admin'
import { verifyAdminToken }    from '@/lib/auth-admin'
import { logAdminAction }      from '@/lib/audit'
import { NextRequest, NextResponse } from 'next/server'

export async function POST(
  request: NextRequest,
  { params }: { params: { userId: string } }
) {
  // Verify admin auth
  const admin = await verifyAdminToken(request)
  if (!admin) return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
  if (!admin.permissions.can_suspend_users) {
    return NextResponse.json({ error: 'Forbidden' }, { status: 403 })
  }

  const { reason } = await request.json()
  if (!reason?.trim()) {
    return NextResponse.json({ error: 'Suspension reason is required' }, { status: 400 })
  }

  const supabase = createSupabaseAdmin()

  // Get current state (for audit log)
  const { data: before } = await supabase
    .from('profiles')
    .select('is_suspended, account_status, suspended_reason')
    .eq('id', params.userId)
    .single()

  // Suspend the user
  const { error } = await supabase
    .from('profiles')
    .update({
      is_suspended:     true,
      account_status:   'suspended',
      suspended_reason: reason.trim(),
      suspended_at:     new Date().toISOString(),
      suspended_by:     admin.id,
      updated_at:       new Date().toISOString(),
    })
    .eq('id', params.userId)

  if (error) {
    return NextResponse.json({ error: error.message }, { status: 500 })
  }

  // Force sign out the user from ALL active sessions
  // This invalidates their Supabase Auth session immediately
  const { error: signOutError } = await supabase.auth.admin.signOut(
    params.userId,
    'global'  // signs out from ALL devices/sessions
  )

  if (signOutError) {
    console.error('Failed to sign out suspended user:', signOutError.message)
    // Don't fail the request — suspension is saved, they'll be blocked on next middleware check
  }

  // Also mark all user_sessions as inactive in our custom sessions table
  await supabase
    .from('user_sessions')
    .update({
      is_active:    false,
      logged_out_at: new Date().toISOString(),
    })
    .eq('user_id', params.userId)
    .eq('is_active', true)

  // Log the admin action
  await logAdminAction({
    adminId:     admin.id,
    action:      'user.suspend',
    targetType:  'user',
    targetId:    params.userId,
    beforeValue: before ?? {},
    afterValue:  { is_suspended: true, account_status: 'suspended', suspended_reason: reason },
    ipAddress:   request.headers.get('x-forwarded-for') ?? undefined,
  })

  return NextResponse.json({ success: true })
}
```

```tsx
  <div className="bg-red-600 text-white text-center py-2 px-4 text-sm font-medium">
    ⚠ MAINTENANCE MODE IS ON — Users are seeing the maintenance page right now.
    <button
      onClick={() => handleMaintenanceToggle(false)}
      className="ml-4 underline hover:no-underline font-semibold"
    >
      Turn it off →
    </button>
  </div>
)}
```
