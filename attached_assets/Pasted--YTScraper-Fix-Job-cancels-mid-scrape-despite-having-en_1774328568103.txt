## FIX 1 — Edge Function timeout (Supabase free tier: 150s hard limit)

The free tier Supabase edge function has a 150-second timeout.
With 500 comments/invocation and reply fetching, this can be hit easily.

**Solution: Reduce PER_INVOCATION_LIMIT from 500 to 200.**

In `supabase/functions/fetch-comments/index.ts`:

```typescript
// Find this line:
const PER_INVOCATION_LIMIT = 500;  // or 2000

// Change to:
const PER_INVOCATION_LIMIT = 200;
// 200 comments per invocation = ~30-60 seconds safely within 150s limit
// The frontend chains more calls automatically — no data loss
```

Also add a safety timeout inside the edge function itself so it exits
cleanly before Deno kills it hard:

```typescript
// At the top of the serve() handler, after parsing the body:
const SAFE_TIMEOUT_MS = 120_000  // 120 seconds — exit cleanly before 150s hard kill

const timeoutPromise = new Promise<never>((_, reject) =>
  setTimeout(() => reject(new Error('SAFE_TIMEOUT')), SAFE_TIMEOUT_MS)
)

// Wrap the entire scraping operation in a race:
let scrapeResult: { totalFetched: number; nextPageToken: string | null }
try {
  scrapeResult = await Promise.race([
    yt.fetchAllComments(
      job.id,
      videoId,
      { ...filters, maxComments: commentTarget },
      commentTarget,
      onBatch,
      startPageToken,
      PER_INVOCATION_LIMIT,
    ),
    timeoutPromise,
  ])
  nextPageToken = scrapeResult.nextPageToken
} catch (timeoutErr: any) {
  if (timeoutErr.message === 'SAFE_TIMEOUT') {
    console.log('[TIMEOUT] Safe timeout reached — saving progress and returning resume token')
    // Get current page token from wherever the YouTubeApi left off
    // Return done=false so frontend chains another call
    const { data: currentJob } = await supabaseAdmin
      .from('jobs')
      .select('next_page_token, downloaded_comments')
      .eq('id', job.id)
      .single()

    // Mark job as still running (not cancelled, not failed)
    await supabaseAdmin.from('jobs').update({
      status: 'running',
      downloaded_comments: alreadyFetched + totalInserted,
    }).eq('id', job.id)

    return json({
      success:       true,
      done:          false,
      nextPageToken: currentJob?.next_page_token ?? null,
      job_id:        job.id,
      video_title:   video.title,
      channel_name:  video.channelName,
      thumbnail:     video.thumbnail,
      comment_count: alreadyFetched + totalInserted,
      credits_used:  creditsUsed,
      balance_after: balance - creditsUsed,
      cancelled:     false,
      timeout:       true,  // flag so frontend knows this was a timeout
    })
  }
  throw timeoutErr  // re-throw non-timeout errors
}
```

## FIX 3 — Frontend polling reading wrong job

In the scrape page component, find the polling useEffect.
Make sure it filters by the SPECIFIC active job ID, not just any job:

```typescript
// WRONG — might read a different job:
const { data } = await supabase
  .from('jobs')
  .select('*')
  .eq('user_id', userId)
  .in('status', ['queued', 'running'])
  .single()

// CORRECT — always poll the specific job we started:
const { data } = await supabase
  .from('jobs')
  .select('*')
  .eq('id', activeJobId)   // ← specific job ID, never user_id filter
  .single()
```

Also fix the completion check — only stop polling when THIS job is done:

```typescript
useEffect(() => {
  if (!activeJobId) return

  const poll = setInterval(async () => {
    const { data: job } = await supabase
      .from('jobs')
      .select('*')
      .eq('id', activeJobId)  // specific job
      .single()

    if (!job) return
    setActiveJob(job)

    console.log(`[POLL] job=${activeJobId} status=${job.status} downloaded=${job.downloaded_comments}`)

    if (job.status === 'completed') {
      clearInterval(poll)
      await refetch()
      setScrapeState('completed')
      toast.success(`✓ ${job.downloaded_comments?.toLocaleString()} comments scraped!`)
    }

    if (job.status === 'cancelled') {
      clearInterval(poll)
      await refetch()
      // Show partial results if we have data, not an error
      if ((job.downloaded_comments ?? 0) >= 100) {
        setScrapeState('completed')
        toast.warning(`Scrape stopped at ${job.downloaded_comments?.toLocaleString()} comments.`)
      } else {
        setScrapeState('failed')
      }
    }

    if (job.status === 'failed') {
      clearInterval(poll)
      await refetch()
      setScrapeState('failed')
      toast.error(job.error_message ?? 'Scrape failed')
    }

    // IMPORTANT: Do NOT stop polling for status='running'
    // Keep polling until a terminal status is reached

  }, 2000)

  return () => clearInterval(poll)
}, [activeJobId])
```

## FIX 6 — Don't set status='cancelled' when credits run out

When `atomic_credit_deduct` returns `{ok: false}`, the job should
stop naturally — but the status should be set to 'completed' (partial)
not 'cancelled'. 'cancelled' means user action.

In the edge function, in the finalisation section:

```typescript
// CURRENT (likely wrong):
if (!isCancelled) {
  if (done) {
    await supabaseAdmin.from('jobs').update({ status: 'completed', ... })
  }
} else {
  // This sets cancelled status even for credit exhaustion!
  await supabaseAdmin.from('jobs').update({ status: 'cancelled', ... })
}

// FIX — track WHY it stopped:
let stopReason: 'completed' | 'cancelled' | 'credits_exhausted' = 'completed'

// In onBatch, when credits run out:
if (!deductResult?.ok) {
  stopReason = 'credits_exhausted'
  return false
}

// In the cancellation check:
if (jobCheck?.status === 'cancelled') {
  isCancelled = true
  stopReason = 'cancelled'
  return false
}

// In finalisation:
if (stopReason === 'completed') {
  await supabaseAdmin.from('jobs').update({
    status: 'completed',
    progress: 100,
    credits_used: creditsUsed,
    downloaded_comments: totalAcrossAllInvocations,
    completed_at: new Date().toISOString(),
  }).eq('id', job.id)
}
else if (stopReason === 'credits_exhausted') {
  // Partial completion — has data but ran out of credits
  await supabaseAdmin.from('jobs').update({
    status: 'completed',          // mark as completed so explorer shows
    progress: 100,
    credits_used: creditsUsed,
    downloaded_comments: totalAcrossAllInvocations,
    completed_at: new Date().toISOString(),
    error_message: 'Stopped: credits exhausted. Showing partial results.',
  }).eq('id', job.id)
}
else if (stopReason === 'cancelled') {
  // User explicitly cancelled
  await supabaseAdmin.from('jobs').update({
    status: 'cancelled',
    downloaded_comments: totalAcrossAllInvocations,
    completed_at: new Date().toISOString(),
  }).eq('id', job.id)
}
```

  id,

  error_message,
