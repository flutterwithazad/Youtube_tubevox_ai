## FIX A — Refresh balance from DB after every scrape completion

This is the most important fix regardless of what diagnostics show.

Find the `useCredits` hook or wherever balance is fetched. Ensure it:
1. Fetches fresh from DB on mount
2. Re-fetches whenever a job completes, fails, or is cancelled
3. Re-fetches when the page regains focus (user switches tabs and comes back)

```typescript
// lib/hooks/useCredits.ts — COMPLETE REWRITE

'use client'
import { useEffect, useState, useCallback, useRef } from 'react'
import { createClient } from '@/lib/supabase/client'

export function useCredits(userId: string | undefined) {
  const [balance, setBalance] = useState<number>(0)
  const [loading, setLoading] = useState(true)
  const supabase = createClient()
  const intervalRef = useRef<NodeJS.Timeout>()

  const fetchBalance = useCallback(async () => {
    if (!userId) return
    try {
      const { data, error } = await supabase
        .from('user_credit_balance')
        .select('balance')
        .eq('user_id', userId)
        .single()

      if (!error && data) {
        console.log('[BALANCE] Fetched from DB:', data.balance)
        setBalance(data.balance ?? 0)
      }
    } catch (err) {
      console.error('[BALANCE] Failed to fetch:', err)
    } finally {
      setLoading(false)
    }
  }, [userId])

  useEffect(() => {
    fetchBalance()

    // Poll every 10 seconds (not 30) so balance stays fresh during active use
    intervalRef.current = setInterval(fetchBalance, 10000)

    // Re-fetch when tab becomes visible again (user switches back)
    const handleVisibilityChange = () => {
      if (document.visibilityState === 'visible') {
        console.log('[BALANCE] Tab became visible — refreshing balance')
        fetchBalance()
      }
    }
    document.addEventListener('visibilitychange', handleVisibilityChange)

    return () => {
      clearInterval(intervalRef.current)
      document.removeEventListener('visibilitychange', handleVisibilityChange)
    }
  }, [fetchBalance])

  return { balance, loading, refetch: fetchBalance }
}
```

## FIX D — Use fresh session token for every edge function call

The JWT token can expire during a long session. Use `getSession()` fresh
for every invocation in the scrape loop, NOT a cached one from the start:

```typescript
// WRONG — token captured once at start, stale after 1 hour:
const { data: { session } } = await supabase.auth.getSession()
const token = session.access_token  // captured once

while (!done) {
  const response = await fetch(edgeFunctionUrl, {
    headers: { 'Authorization': `Bearer ${token}` }  // may be expired by page 5
  })
}

// CORRECT — fresh token on every invocation:
while (!done) {
  // Get fresh session every time
  const { data: { session } } = await supabase.auth.getSession()
  if (!session) {
    onError('Session expired. Please refresh the page.')
    return
  }

  const response = await fetch(edgeFunctionUrl, {
    method: 'POST',
    headers: {
      'Content-Type':  'application/json',
      'Authorization': `Bearer ${session.access_token}`,  // always fresh
      'apikey':        process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    },
    body: JSON.stringify(body),
  })

  // ... handle response
}
```

## FIX F — Clean up orphaned reservations from past jobs

Run this SQL in Supabase to refund any past jobs where credits were reserved
but never settled:

```sql
-- Find jobs with orphaned reservations
WITH job_ledger AS (
  SELECT
    j.id          AS job_id,
    j.user_id,
    j.status,
    j.credits_reserved,
    j.downloaded_comments,
    COALESCE(SUM(CASE WHEN cl.source_type IN ('job_settle','job_cancel_refund','refund') THEN cl.amount ELSE 0 END), 0) AS already_refunded,
    COALESCE(SUM(CASE WHEN cl.source_type = 'job_reserve' THEN ABS(cl.amount) ELSE 0 END), 0) AS total_reserved
  FROM public.jobs j
  LEFT JOIN public.credit_ledger cl ON cl.source_id = j.id
  WHERE j.status IN ('completed','cancelled','failed')
  GROUP BY j.id, j.user_id, j.status, j.credits_reserved, j.downloaded_comments
)
SELECT
  job_id,
  user_id,
  status,
  credits_reserved,
  downloaded_comments,
  total_reserved,
  already_refunded,
  (total_reserved + already_refunded) AS net_deducted,
  downloaded_comments AS should_have_deducted,
  (total_reserved + already_refunded - downloaded_comments) AS over_charged
FROM job_ledger
WHERE (total_reserved + already_refunded - COALESCE(downloaded_comments, 0)) > 0
ORDER BY over_charged DESC;
```

For each row where `over_charged > 0`, insert a correction:

```sql
-- Fix one over-charged job (replace values with actual query results):
INSERT INTO public.credit_ledger (user_id, amount, source_type, source_id, description)
VALUES (
  '<user_id>',
  <over_charged_amount>,     -- positive number = credits back
  'refund',
  '<job_id>',
  'Correction: over-charged credits from orphaned reservation'
);
```
